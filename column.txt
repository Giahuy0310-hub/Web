-- Tạo cơ sở dữ liệu "project"
CREATE DATABASE project9;

-- Sử dụng cơ sở dữ liệu "project"
USE project9;

-- Tạo bảng "product_id"
CREATE TABLE product_id (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_product CHAR(20) UNIQUE
);

-- Tạo bảng "categories"
CREATE TABLE categories (
    id_dm INT AUTO_INCREMENT PRIMARY KEY,
    TenDanhMuc NVARCHAR(255)
);

-- Tạo bảng "color"
CREATE TABLE color (
    id_color INT AUTO_INCREMENT PRIMARY KEY,
    tenmau NVARCHAR(255),
    hex_color VARCHAR(7) -- Ví dụ: Mã màu HEX (#RRGGBB)
);

-- Tạo bảng "products"
CREATE TABLE products (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_product INT,
    id_dm INT,
    id_color INT,
    ten_san_pham NVARCHAR(255),
    link_hinh_anh TEXT,
    loaisanpham nchar(20),
    gia INT,
    img1 TEXT,
    img2 TEXT,
    img3 TEXT,
    img4 TEXT,
    rating DECIMAL(3, 2),
    so_danh_gia INT,
    FOREIGN KEY (id_color) REFERENCES color(id_color),
    FOREIGN KEY (id_product) REFERENCES product_id(id_product),
    FOREIGN KEY (id_dm) REFERENCES categories(id_dm)
);

-- Tạo bảng "product_color"
CREATE TABLE product_color (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_product INT,
    id_color INT,
    FOREIGN KEY (id_product) REFERENCES product_id(id),
    FOREIGN KEY (id_color) REFERENCES color(id_color)
);

-- Tạo bảng "login"
CREATE TABLE login (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fullname NVARCHAR(255),
    phone_number VARCHAR(20),
    email VARCHAR(255),
    password VARCHAR(255)
);

CREATE TABLE DonHang (
    id INT AUTO_INCREMENT PRIMARY KEY,
    hoten VARCHAR(255),
    sodienthoai VARCHAR(15),
    email VARCHAR(255),
    sonha_duong VARCHAR(255),
    phuong_xa VARCHAR(100),
    quan_huyen VARCHAR(100),
    tinh_thanh VARCHAR(100),
    ghichu TEXT
);


CREATE TABLE ChiTietDonHang (
    id INT AUTO_INCREMENT PRIMARY KEY,
    donhang_id INT,
    sanpham_id INT,
    so_luong INT,
    FOREIGN KEY (donhang_id) REFERENCES DonHang(id),
    FOREIGN KEY (sanpham_id) REFERENCES Products(id_product)
);




CREATE TABLE giohang (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_product int ,
    ten_san_pham text,
    id_color int,
    nguoidung_id INT,
    gia DECIMAL(10, 2), 
    link_hinh_anh TEXT,
    quantity INT,
    size char(10),
    FOREIGN KEY (id_product) REFERENCES product_id(id_product),
    FOREIGN KEY (nguoidung_id) REFERENCES login(id),
    FOREIGN KEY (id_color) REFERENCES color(id_color)
);


CREATE TABLE `province` (
  `province_id` int NOT NULL,
  `name` varchar(64) COLLATE utf8mb4_general_ci NOT NULL
) ENGINE=MyISAM DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='Tỉnh thành';

CREATE TABLE `district` (
  `district_id` int NOT NULL,
  `province_id` int NOT NULL,
  `name` varchar(64) COLLATE utf8mb4_general_ci NOT NULL
) ENGINE=MyISAM DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='Quận huyện';

CREATE TABLE `wards` (
  `wards_id` int NOT NULL,
  `district_id` int NOT NULL,
  `name` varchar(64) COLLATE utf8mb4_general_ci NOT NULL
) ENGINE=MyISAM DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='Xã Phường';

ALTER TABLE district ADD FOREIGN KEY (`province_id`) REFERENCES province(`province_id`);
ALTER TABLE wards ADD FOREIGN KEY (`district_id`) REFERENCES district(`district_id`);
